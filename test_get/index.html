    class TransitGetTest(TestCase):

    @classmethod
    def setUpTestData(cls):
        Transit.objects.create(destination='Уфа')

    def test_get_transit(self):
        url = reverse('airport_admin:transit')

        data = {"count": 1, "next": None, "previous": None, "results": [{"id": 2, "destination": "Уфа"}]}

        response = self.client.get(url, format ='json')
        self.assertEqual(response.status_code, status.HTTP_200_OK)
        self.assertEqual(response.json(), data)

    class FlightAsScheduledGetTest(TestCase):
